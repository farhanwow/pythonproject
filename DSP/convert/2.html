#Klasifikasi sinyal
#sinyal nyata dan kompleks
import numpy as np
import matplotlib.pyplot as plt

# Parameter
a = -5 # rentang waktu
b = 5
dt = 0.1
sigma = 0  # s = 0 +0j
omega = 0

# PROSEDUR
t = np.arange(a, b, dt)
s = complex(sigma,omega)
sinyal = np.exp(s*t)

# OUTPUT - grafik
plt.figure(1)
plt.plot(t,np.real(sinyal),label='real')
plt.plot(t,np.imag(sinyal),label='imagijiner')
plt.xlabel('t')
plt.ylabel('x(t)')
plt.legend()
plt.margins(0.1)
plt.title('Sinyal  e^('+str(sigma) +' + '+str(omega)+'j)')
plt.show()
